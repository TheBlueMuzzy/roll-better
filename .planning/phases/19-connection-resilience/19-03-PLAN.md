**ConnectionToast.tsx â€” New component:**
Create a toast notification component that auto-dismisses:

```tsx
import { useEffect } from 'react';
import { useGameStore } from '../store/gameStore';

export function ConnectionToasts() {
  const toasts = useGameStore((s) => s.connectionToasts);
  const removeToast = useGameStore((s) => s.removeConnectionToast);

  return (
    <div className="connection-toasts">
      {toasts.map((toast) => (
        <ConnectionToastItem key={toast.id} toast={toast} onDismiss={() => removeToast(toast.id)} />
      ))}
    </div>
  );
}

function ConnectionToastItem({ toast, onDismiss }: { toast: { id: string; message: string; type: string }; onDismiss: () => void }) {
  useEffect(() => {
    const timer = setTimeout(onDismiss, 3000);
    return () => clearTimeout(timer);
  }, [onDismiss]);

  return (
    <div className={`connection-toast connection-toast-${toast.type}`}>
      {toast.message}
    </div>
  );
}
```
